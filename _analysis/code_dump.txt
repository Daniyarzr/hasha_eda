0036:             ->when($categoryId, fn ($query) => $query->where('category_id', $categoryId))
0037:             ->when($search !== '', function ($query) use ($search) {
0038:                 $query->where(function ($searchQuery) use ($search): void {
0039:                     $searchQuery
0040:                         ->where('name', 'like', '%' . $search . '%')
0041:                         ->orWhere('description', 'like', '%' . $search . '%')
0042:                         ->orWhereHas('restaurant', fn ($restaurantQuery) => $restaurantQuery->where('name', 'like', '%' . $search . '%'));
0043:                 });

0046:             ->when(is_numeric($priceTo), fn ($query) => $query->where('price', '<=', (float) $priceTo));
0047: 
0048:         if ($sort === 'price_asc') {
0049:             $dishesQuery->orderBy('price');
0050:         } elseif ($sort === 'price_desc') {
0051:             $dishesQuery->orderByDesc('price');
0052:         } elseif ($sort === 'new') {
0053:             $dishesQuery->orderByDesc('id');
0054:         } else {
0055:             $dishesQuery->orderByDesc('id');
0056:         }
0057: 
0058:         $dishes = $dishesQuery->get();

0060:         return view('catalog.index', [
0061:             'dishes' => $dishes,
0062:             'restaurants' => $restaurants,
0063:             'categories' => $categories,
0064:             'filters' => [

0046:         ]);
0047: 
0048:         $dish = Dish::query()->where('is_available', true)->findOrFail($validated['dish_id']);
0049:         $quantity = $validated['quantity'] ?? 1;

0058:             $item->quantity = min(99, $item->quantity + $quantity);

0060:             $item->quantity = $quantity;
0061:         }
0062: 
0063:         $item->save();
0064: 
0065:         return back()->with('success', 'Блюдо добавлено в корзину.');
0066:     }

0018:         $user = $request->user();
0019:         $cartItems = $user->cartItems()->with('dish.restaurant')->get();
0020: 
0021:         if ($cartItems->isEmpty()) {

0026: 
0027:         if ($itemsByRestaurant->isEmpty()) {
0028:             return redirect()->route('cart.index')->with('error', 'Не удалось оформить заказ: блюда недоступны.');
0029:         }
0030: 
0031:         DB::transaction(function () use ($user, $itemsByRestaurant, $validated, $cartItems): void {
0032:             
0033:             $this->createOrdersFromCart($user, $itemsByRestaurant, $validated);
0034: 
0035:           
0036:             $this->updateUserContacts($user, $validated);
0037: 
0038:            
0039:             $this->clearCart($cartItems);
0040:         });
0041: 
0042:         return redirect()->route('account.index')->with('success', 'Заказ успешно оформлен.');
0043:     }

0045:     private function validateCheckoutData(Request $request): array
0046:     {
0047:         return $request->validate([
0048:             'delivery_address' => ['required', 'string', 'max:255'],
0049:             'customer_phone' => ['required', 'string', 'max:50'],
0050:             'comment' => ['nullable', 'string', 'max:1000'],
0051:         ]);

0053: 
0054:     private function groupItemsByRestaurant(Collection $cartItems): Collection
0055:     {
0056:         return $cartItems
0057:             ->filter(fn (Cart $item) => $item->dish && $item->dish->restaurant)
0058:             ->groupBy(fn (Cart $item) => $item->dish->restaurant->id);
0059:     }
0060: 
0061:     private function createOrdersFromCart(User $user, Collection $itemsByRestaurant, array $validated): void
0062:     {
0063:         foreach ($itemsByRestaurant as $restaurantId => $restaurantItems) {
0064:             $restaurant = $restaurantItems->first()->dish->restaurant;
0065:             $deliveryFee = (float) $restaurant->delivery_fee;
0066: 
0067:             $subtotal = 0.0;
0068:             foreach ($restaurantItems as $item) {
0069:                 $subtotal += (float) $item->price * $item->quantity;
0070:             }
0071: 
0072:             $order = $this->createOrder(
0073:                 $user,
0074:                 (int) $restaurantId,
0075:                 $validated,
0076:                 $subtotal,
0077:                 $deliveryFee
0078:             );
0079: 
0080:             $this->createOrderItems($order, $restaurantItems);
0081:         }
0082:     }
0083: 
0084:     private function createOrder(
0085:         User $user,
0086:         int $restaurantId,
0087:         array $validated,
0088:         float $subtotal,
0089:         float $deliveryFee
0090:     ): Order {
0091:         return $user->orders()->create([
0092:             'restaurant_id' => $restaurantId,
0093:             'status' => 'new',
0094:             'delivery_address' => $validated['delivery_address'],
0095:             'customer_phone' => $validated['customer_phone'],
0096:             'comment' => $validated['comment'] ?? null,
0097:             'total_amount' => $subtotal + $deliveryFee,
0098:             'delivery_fee' => $deliveryFee,
0099:             'ordered_at' => now(),
0100:         ]);
0101:     }
0102: 
0103:     private function createOrderItems(Order $order, Collection $restaurantItems): void
0104:     {
0105:         foreach ($restaurantItems as $item) {
0106:             $order->items()->create([
0107:                 'dish_id' => $item->dish_id,
0108:                 'dish_name' => $item->dish?->name ?? 'Неизвестное блюдо',
0109:                 'quantity' => $item->quantity,
0110:                 'price' => $item->price,
0111:             ]);
0112:         }
0113:     }
0114: 
0115:     private function updateUserContacts(User $user, array $validated): void
0116:     {
0117:         $user->update([
0118:             'phone' => $validated['customer_phone'],
0119:             'default_address' => $validated['delivery_address'],
0120:         ]);
0121:     }
0122: 
0123:     private function clearCart(Collection $cartItems): void
0124:     {
0125:         Cart::query()->whereIn('id', $cartItems->pluck('id'))->delete();
0126:     }
0127: }
